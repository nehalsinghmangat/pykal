\begin{minipage}{7cm}
  \begin{align*}
    \dot{\mathbf{x}} &=
    \begin{bmatrix}
      \dot{x}_1 \\
      \dot{x}_2 \\
      \dot{x}_3 \\
      \dot{x}_4
    \end{bmatrix}
    =
    \begin{bmatrix}
      x_2 \\
      f_1(x_1, x_2, x_3, x_4) \\
      x_4 \\
      f_2(x_1, x_2, x_3, x_4)
    \end{bmatrix} \\
    \mathbf{y} &=
    \begin{bmatrix}
      x_1 \\
      x_3
    \end{bmatrix}
  \end{align*}

Let $\Delta = x_3 - x_1$:

\begin{align*}
f_1(x) &= -\frac{g}{\ell_1} \sin(x_1) + \frac{k}{m_1 \ell_1^2} \Delta \\
f_2(x) &= -\frac{g}{\ell_2} \sin(x_3) - \frac{k}{m_2 \ell_2^2} \Delta
\end{align*}
\end{minipage}
